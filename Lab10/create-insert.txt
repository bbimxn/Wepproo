const sql = `CREATE TABLE products (
    product_id INTEGER PRIMARY KEY,
    name VARCHAR(255),
    description TEXT,
    price DECIMAL(10, 2),
    quantity INTEGER,
    image_url VARCHAR(255)
);`;

db.run(sql, (err) => {
    if (err) {
        return console.error('Error creating table:', err.message);
    }
    console.log('Table created successful');
});
